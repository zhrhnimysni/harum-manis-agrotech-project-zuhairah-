    <script>
        // Deklarasi variabel global parameter model matematika
        let L = 420;
        let k = 0.08;
        let t0 = 50;
        let myChart = null;

        // Inisialisasi konfigurasi graf Chart.js
        function initChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');
            
            // Menghasilkan titik data dari minggu ke-0 hingga ke-12
            const dataPoints = generatePlotData();

            myChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: dataPoints.labels,
                    datasets: [
                        {
                            label: 'Berat Buah f(t) (gram)',
                            data: dataPoints.weight,
                            borderColor: '#10b981',
                            borderWidth: 3,
                            backgroundColor: 'rgba(16, 185, 129, 0.03)',
                            fill: true,
                            yAxisID: 'y',
                            tension: 0.2,
                            pointRadius: 0
                        },
                        {
                            label: "Kadar Pertumbuhan f'(t) (g/hari)",
                            data: dataPoints.rate,
                            borderColor: '#fbbf24',
                            borderWidth: 2,
                            borderDash: [4, 4],
                            backgroundColor: 'transparent',
                            yAxisID: 'y1',
                            tension: 0.2,
                            pointRadius: 0
                        },
                        {
                            label: 'Kedudukan Hari Semasa',
                            data: [], // Diisi secara dinamis berdasarkan pergerakan slider
                            borderColor: '#ef4444',
                            borderWidth: 1,
                            pointBackgroundColor: '#ef4444',
                            pointRadius: 5,
                            showLine: true
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        x: {
                            grid: { color: 'rgba(255, 255, 255, 0.02)' },
                            ticks: { color: '#94a3b8', font: { size: 10 } },
                            title: { display: true, text: 'Hari Selepas Antesis (DAA)', color: '#64748b', font: { size: 10 } }
                        },
                        y: {
                            type: 'linear',
                            display: true,
                            position: 'left',
                            grid: { color: 'rgba(255, 255, 255, 0.03)' },
                            ticks: { color: '#10b981', font: { size: 10 } },
                            title: { display: true, text: 'Berat Kumulatif (g)', color: '#10b981', font: { size: 10 } }
                        },
                        y1: {
                            type: 'linear',
                            display: true,
                            position: 'right',
                            grid: { drawOnChartArea: false },
                            ticks: { color: '#fbbf24', font: { size: 10 } },
                            title: { display: true, text: 'Kadar Pembesaran (g/hari)', color: '#fbbf24', font: { size: 10 } }
                        }
                    },
                    plugins: {
                        legend: {
                            position: 'top',
                            labels: { color: '#94a3b8', boxWidth: 12, font: { size: 11 } }
                        }
                    }
                }
            });
        }

        // Penghitungan Model Logistik Matematika
        function evaluateLogistic(t) {
            return L / (1 + Math.exp(-k * (t - t0)));
        }

        // Penghitungan Terbitan Pertama (Kadar Pertumbuhan)
        function evaluateDerivative(t) {
            const expTerm = Math.exp(-k * (t - t0));
            return (L * k * expTerm) / Math.pow(1 + expTerm, 2);
        }

        // Menghasilkan array data berkelanjutan untuk grafik
        function generatePlotData() {
            const labels = [];
            const weight = [];
            const rate = [];
            
            // Loop dari hari ke-0 hingga ke-84 (12 minggu)
            for(let day = 0; day <= 84; day += 2) {
                labels.push(day);
                weight.push(evaluateLogistic(day));
                rate.push(evaluateDerivative(day));
            }
            return { labels, weight, rate };
        }

        // Merender ulang graf apabila parameter diubah
        function renderChart() {
            if(!myChart) return;
            const newData = generatePlotData();
            myChart.data.datasets[0].data = newData.weight;
            myChart.data.datasets[1].data = newData.rate;
            myChart.update('none'); // Update graf dengan transisi lancar tanpa animasi ulang total
        }

        // Generator teks tafsiran awam perkembangan buah
        function getLaymanInterpretation(day) {
            if (day < 20) {
                return "🟢 <b>Fasa Awal (Pembahagian Sel):</b> Buah baru terbentuk. Pertumbuhan agak perlahan pada peringkat awal sel aktif membahagi.";
            } else if (day >= 20 && day < 45) {
                return "⚡ <b>Fasa Eksponen (Pemanjangan Sel):</b> Struktur buah mula membesar dengan pantas. Kadar penyerapan air & nutrien sangat tinggi.";
            } else if (day >= 45 && day <= 55) {
                return "🔥 <b>Titik Inflexion (Kadar Maksimum):</b> Buah membesar pada kelajuan paling maksimum! Fasa kritikal penentuan saiz gred premium.";
            } else if (day > 55 && day < 75) {
                return "📈 <b>Fasa Penurunan Kadar:</b> Berat terus meningkat tetapi momentumnya mula perlahan kerana isi buah sudah mula padat.";
            } else {
                return "🏆 <b>Fasa Matang Pemantapan:</b> Berat buah mencapai tahap kestabilan maksimum (asymptote). Kulit & isi sedia dituai.";
            }
        }

        // Mengemas kini papan pemuka data secara dinamik
        function updateDashboard(weekValue) {
            const currentDay = parseFloat(weekValue) * 7;
            
            // Hitung nilai berat dan kadar semasa
            const wVal = evaluateLogistic(currentDay);
            const rVal = evaluateDerivative(currentDay);

            // Perbarui elemen teks HTML
            document.getElementById('selectedWeekTxt').innerText = `Minggu ${parseFloat(weekValue).toFixed(1)}`;
            document.getElementById('selectedDaaBadge').innerText = `HARI ${currentDay.toFixed(0)} DAA`;
            
            document.getElementById('dispWeight').textContent = `${wVal.toFixed(1)}g`;
            document.getElementById('dispRate').textContent = `${rVal.toFixed(2)}g/hari`;
            document.getElementById('laymanExplanation').innerHTML = getLaymanInterpretation(currentDay);

            // Tambah penanda titik semasa pada garisan graf
            if(myChart) {
                myChart.data.datasets[2].data = [{x: currentDay, y: wVal}];
                myChart.update('none');
            }

            // Kemas kini paparan Math LaTeX di dalam dokumen
            if (typeof renderMathInElement === 'function') {
                renderMathInElement(document.body);
            }
        }

        // Event listener untuk penyesuaian parameter (Kalibrasi)
        const sliderL = document.getElementById('sliderL');
        const sliderK = document.getElementById('sliderK');
        const sliderT0 = document.getElementById('sliderT0');

        sliderL.addEventListener('input', (e) => {
            L = parseInt(e.target.value);
            document.getElementById('txtL').textContent = `${L}g`;
            document.getElementById('valL').textContent = `${L}g`;
            renderChart();
            updateDashboard(document.getElementById('weekSlider').value);
        });

        sliderK.addEventListener('input', (e) => {
            k = parseFloat(e.target.value);
            document.getElementById('txtK').textContent = k.toFixed(3);
            document.getElementById('valK').textContent = k.toFixed(3);
            renderChart();
            updateDashboard(document.getElementById('weekSlider').value);
        });

        sliderT0.addEventListener('input', (e) => {
            t0 = parseInt(e.target.value);
            document.getElementById('txtT0').textContent = `${t0} DAA`;
            document.getElementById('valT0').textContent = `${t0} DAA`;
            renderChart();
            updateDashboard(document.getElementById('weekSlider').value);
        });

        // Tombol reset parameter ke konfigurasi awal
        document.getElementById('resetParamsBtn').addEventListener('click', () => {
            L = 420;
            k = 0.080;
            t0 = 50;
            sliderL.value = L;
            sliderK.value = k;
            sliderT0.value = t0;
            document.getElementById('txtL').textContent = '420g';
            document.getElementById('valL').textContent = '420g';
            document.getElementById('txtK').textContent = '0.080';
            document.getElementById('valK').textContent = '0.080';
            document.getElementById('txtT0').textContent = '50 DAA';
            document.getElementById('valT0').textContent = '50 DAA';
            renderChart();
            updateDashboard(document.getElementById('weekSlider').value);
        });

        // ========================================================
        // LOGIK KOD ANIMASI AUTOMATIK (SINKRONISASI TINGGI)
        // ========================================================
        let animationInterval = null;
        const weekSlider = document.getElementById('weekSlider');
        const playButton = document.getElementById('playButton');
        const playIcon = document.getElementById('playIcon');
        const playText = document.getElementById('playText');

        function toggleAnimation() {
            if (animationInterval) {
                // Hentikan animasi (Pause)
                clearInterval(animationInterval);
                animationInterval = null;
                playIcon.innerText = "▶";
                playText.innerText = "Mainkan Animasi";
                playButton.className = "bg-emerald-500 hover:bg-emerald-600 text-slate-950 font-bold text-xs px-4 py-2 rounded-xl flex items-center gap-1.5 transition-all cursor-pointer shadow-lg shadow-emerald-500/10";
            } else {
                // Jalankan animasi (Play)
                playIcon.innerText = "⏸";
                playText.innerText = "Hentikan Animasi";
                playButton.className = "bg-yellow-500 hover:bg-yellow-600 text-slate-950 font-bold text-xs px-4 py-2 rounded-xl flex items-center gap-1.5 transition-all cursor-pointer shadow-lg shadow-yellow-500/10";
                
                // Mulakan dari minggu 0 semula sekiranya kedudukan semasa di akhir garisan masa
                if (parseFloat(weekSlider.value) >= 12) {
                    weekSlider.value = 0;
                }

                animationInterval = setInterval(() => {
                    let currentWeek = parseFloat(weekSlider.value);
                    currentWeek += 0.1; // Menambah selang masa sebanyak 0.1 minggu
                    
                    if (currentWeek <= 12) {
                        weekSlider.value = currentWeek.toFixed(1);
                        updateDashboard(currentWeek);
                    } else {
                        // Tamat secara automatik pada minggu ke-12
                        toggleAnimation();
                    }
                }, 100); // Bergerak setiap 100 milisaat
            }
        }

        // Pengaktifan kawalan animasi
        if (playButton) {
            playButton.addEventListener('click', toggleAnimation);
        }

        // Apabila pengguna melaras slider secara manual, hentikan simulasi automatik seketika
        if (weekSlider) {
            weekSlider.addEventListener('input', (e) => {
                if (animationInterval) toggleAnimation(); 
                updateDashboard(e.target.value);
            });
        }

        // Permulaan aplikasi sebaik sahaja tetingkap dimuatkan sepenuhnya
        window.onload = function() {
            if (typeof renderMathInElement === 'function') {
                renderMathInElement(document.body);
            }
            initChart();
            
            // Set kedudukan permulaan yang optimum pada titik tengah (inflection point)
            if (weekSlider) {
                weekSlider.value = 6.0;
            }
            updateDashboard(6.0);
            
            if (typeof lucide !== 'undefined' && lucide.createIcons) {
                lucide.createIcons();
            }
        };
    </script>
