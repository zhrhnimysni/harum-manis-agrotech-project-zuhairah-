        // Constants for the Harumanis growth model
        const L = 450;    // Maximum asymptotic weight (grams)
        const k = 0.65;   // Growth rate constant
        const t0 = 6.0;   // Week of maximum growth (inflection point)

        // Mathematical functions
        // f(t) = Cumulative Weight
        function getWeight(t) {
            return L / (1 + Math.exp(-k * (t - t0)));
        }

        // f'(t) = Instantaneous growth rate
        function getDerivative(t) {
            const expTerm = Math.exp(-k * (t - t0));
            const denominator = Math.pow(1 + expTerm, 2);
            return (L * k * expTerm) / denominator;
        }

        // Biological assessment dictionary matching different developmental periods

                <span class="text-yellow-400 font-bold">${grade}</span>). 
                <br><br>
                Its instantaneous growth rate is <span class="text-yellow-400 font-bold">${rate.toFixed(2)} grams per week</span>. 
                In plain terms, <strong class="text-slate-100">${insight.desc}</strong>
            `;
        }

        function toggleAnimation() {
            if (animationInterval) {
                // Jika sedang berjalan, kita PAUSE
                clearInterval(animationInterval);
                animationInterval = null;
                playIcon.innerText = "▶";
                playText.innerText = "Mainkan Animasi";
